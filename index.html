💡 Why are we doing this? Hackers don't steal plaintext passwords; they steal lists of hashes from databases. If simple, raw algorithms like SHA-256 are used, hackers can test billions of guesses per second to reverse-engineer weak patterns. To neutralize this threat, modern cybersecurity uses salting (concatenating random salts to deter lookup files) and key stretching (utilizing slow hashing functions like bcrypt which waste CPU time to prevent automated GPU lookups).

    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    email VARCHAR(255) UNIQUE NOT NULL,
    username VARCHAR(50) UNIQUE NOT NULL,

    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,

    lockout_until TIMESTAMP WITH TIME ZONE
);

Node.js / Express Auth Route Implementations

// Express.js & bcryptjs Password Authentication Middleware
const express = require('express');
const bcrypt = require('bcryptjs'); // standard asynchronous bcrypt library
const jwt = require('jsonwebtoken'); // token provision tool
const app = express();

app.use(express.json()); // JSON payload body parser

// 1. SIGNUP / REGISTER ROUTE
app.post('/api/auth/register', async (req, res) => {
    try {
        const { email, username, password } = req.body;
        
        // Input validation & sanitization
        if (!email || !username || !password) {
            return res.status(400).json({ error: 'All fields are strictly required.' });
        }
        
        // Basic length validation
        if (password.length < 8) {
            return res.status(400).json({ error: 'Password candidate does not meet minimum length requirement (8).' });
        }

        // Generate dynamic salt. Cost factor set to 12 (standard for production workload balance)
        const saltRounds = 12; 
        const salt = await bcrypt.genSalt(saltRounds);
        
        // Generate cryptographic salted check digest (takes approx ~100-250ms per digest)
        const hashedPassword = await bcrypt.hash(password, salt);

        // Save safely into database - raw password is NEVER stored or logged!
        const newUser = await db.query(
            'INSERT INTO users (email, username, password_hash) VALUES ($1, $2, $3) RETURNING id, email, created_at',
            [email, username, hashedPassword]
        );

        res.status(201).json({ 
            message: 'User account registered securely.', 
            userId: newUser.rows[0].id 
        });
        
    } catch (err) {
        console.error('Registration processing error:', err.message);
        res.status(500).json({ error: 'Internal secure server exception occurred.' });
    }
});

// 2. SIGNIN / IDENTITY VERIFICATION ROUTE
app.post('/api/auth/login', async (req, res) => {
    try {
        const { email, password } = req.body;
        
        if (!email || !password) {
            return res.status(400).json({ error: 'Email and password candidates are required.' });
        }

        // Locate candidate profile
        const userResult = await db.query('SELECT * FROM users WHERE email = $1', [email]);
        if (userResult.rows.length === 0) {
            // Apply constant-time processing or generic feedback to deter timing attack vector
            return res.status(401).json({ error: 'Invalid identification credentials.' });
        }
        
        const user = userResult.rows[0];

        // Decrypt / Compare password safely using bcrypt standard math library which avoids timing attacks
        const isMatch = await bcrypt.compare(password, user.password_hash);
        
        if (!isMatch) {
            return res.status(401).json({ error: 'Invalid identification credentials.' });
        }

        // Provision Secure Web Tokens
        const token = jwt.sign(
            { id: user.id, email: user.email }, 
            process.env.JWT_SECRET_KEY, 
            { expiresIn: '2h' }
        );

        res.json({ message: 'Authentication successful.', token });

    } catch (err) {
        console.error('Login processing error:', err.message);
        res.status(500).json({ error: 'Internal secure server exception occurred.' });
    }
});
